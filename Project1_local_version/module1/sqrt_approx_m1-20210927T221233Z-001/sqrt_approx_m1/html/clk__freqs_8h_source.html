25//Get the peripheral bus clock frequency
26static inline uint32_t bus_frequency(void) {
27 return SystemCoreClock / (((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV4_MASK) >> SIM_CLKDIV1_OUTDIV4_SHIFT) + 1);
28}

30//Get external oscillator (crystal) frequency
31static uint32_t extosc_frequency(void) {
32 uint32_t MCGClock = SystemCoreClock * (1u + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV1_MASK) >> SIM_CLKDIV1_OUTDIV1_SHIFT));
33
34 if ((MCG->C1 & MCG_C1_CLKS_MASK) == MCG_C1_CLKS(2)) //MCG clock = external reference clock
35 return MCGClock;
36
37 uint32_t divider, multiplier;
38 #ifdef MCG_C5_PLLCLKEN0_MASK //PLL available
39 if ((MCG->C1 & MCG_C1_CLKS_MASK) == MCG_C1_CLKS(0)) { //PLL/FLL is selected
40 if ((MCG->C6 & MCG_C6_PLLS_MASK) == 0x0u) { //FLL is selected
41 #endif
42 if ((MCG->S & MCG_S_IREFST_MASK) == 0x0u) { //FLL uses external reference
43 divider = (uint8_t)(1u << ((MCG->C1 & MCG_C1_FRDIV_MASK) >> MCG_C1_FRDIV_SHIFT));
44 if ((MCG->C2 & MCG_C2_RANGE0_MASK) != 0x0u)
45 divider <<= 5u;
46 /* Select correct multiplier to calculate the MCG output clock */
47 switch (MCG->C4 & (MCG_C4_DMX32_MASK | MCG_C4_DRST_DRS_MASK)) {
48 case 0x0u:
49 multiplier = 640u;
50 break;
51 case 0x20u:
52 multiplier = 1280u;
53 break;
54 case 0x40u:
55 multiplier = 1920u;
56 break;
57 case 0x60u:
58 multiplier = 2560u;
59 break;
60 case 0x80u:
61 multiplier = 732u;
62 break;
63 case 0xA0u:
64 multiplier = 1464u;
65 break;
66 case 0xC0u:
67 multiplier = 2197u;
68 break;
69 case 0xE0u:
70 default:
71 multiplier = 2929u;
72 break;
73 }
74
75 return MCGClock * divider / multiplier;
76 }
77 #ifdef MCG_C5_PLLCLKEN0_MASK
78 } else { //PLL is selected
79 divider = (1u + (MCG->C5 & MCG_C5_PRDIV0_MASK));
80 multiplier = ((MCG->C6 & MCG_C6_VDIV0_MASK) + 24u);
81 return MCGClock * divider / multiplier;
82 }
83 }
84 #endif
85
86 //In all other cases either there is no crystal or we cannot determine it
87 //For example when the FLL is running on the internal reference, and there is also an
88 //external crystal. However these are unlikely situations
89 return 0;
90}

92//Get MCG PLL/2 or FLL frequency, depending on which one is active, sets PLLFLLSEL bit
93static uint32_t mcgpllfll_frequency(void) {
94 if ((MCG->C1 & MCG_C1_CLKS_MASK) != MCG_C1_CLKS(0)) //PLL/FLL is not selected
95 return 0;
96
97 uint32_t MCGClock = SystemCoreClock * (1u + ((SIM->CLKDIV1 & SIM_CLKDIV1_OUTDIV1_MASK) >> SIM_CLKDIV1_OUTDIV1_SHIFT));
98 #ifdef MCG_C5_PLLCLKEN0_MASK
99 if ((MCG->C6 & MCG_C6_PLLS_MASK) == 0x0u) { //FLL is selected
100 SIM->SOPT2 &= ~SIM_SOPT2_PLLFLLSEL_MASK; //MCG peripheral clock is FLL output
101 #endif
102 return MCGClock;
103 #ifdef MCG_C5_PLLCLKEN0_MASK
104 } else { //PLL is selected
105 SIM->SOPT2 |= SIM_SOPT2_PLLFLLSEL_MASK; //MCG peripheral clock is PLL output
106 return (MCGClock >> 1);
107 }
108 #endif
109
110 //It is possible the SystemCoreClock isn't running on the PLL, and the PLL is still active
111 //for the peripherals, this is however an unlikely setup
112}

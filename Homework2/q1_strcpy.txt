#include <stdio.h>
#include <string.h>

int mystrcpy(char str1[], const char str2[]){
    int len1;
    int len2;
    
    for(len1 = 0; str1[len1] !='\0'; len1++);
    for(len2 = 0; str2[len2] !='\0'; len2++);
    
    if(len1 < len2){
        printf("destination string size is not enough to accomodate source string\n");
        return -1;
    }
    while(*str2 != '\0'){
        *str1++=*str2++;
    }
}

int main()
{
    printf("Hello World");
    char str1[] = "abc";
    char str2[] = "pqrxyz";
    
    printf("\nbefore copy:\nstr1 = %s\nstr2 = %s\n\n", str1, str2);
    
    if(mystrcpy(str1,str2)!=0)
        printf("copy failed\n");
    else{
        printf("after copy:\nstr1 = %s\nstr2 = %s\n", str1, str2);
    }
    return 0;
}
